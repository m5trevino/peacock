# websocket_server.py
import asyncio
import websockets
import subprocess
import json

async def handle_command(websocket, path):
    async for message in websocket:
        data = json.loads(message)
        command = data['command']
        
        # Execute command safely
        try:
            result = subprocess.run(
                command.split(), 
                capture_output=True, 
                text=True, 
                timeout=30
            )
            
            response = {
                'output': result.stdout,
                'error': result.stderr,
                'returncode': result.returncode
            }
        except Exception as e:
            response = {'error': str(e)}
            
        await websocket.send(json.dumps(response))

# Start WebSocket server
start_server = websockets.serve(handle_command, "localhost", 8765)

# terminal_api.py
from flask import Flask, request, jsonify
import subprocess

app = Flask(__name__)

@app.route('/execute', methods=['POST'])
def execute_command():
    command = request.json['command']
    
    try:
        result = subprocess.run(
            command.split(),
            capture_output=True,
            text=True,
            timeout=30
        )
        
        return jsonify({
            'output': result.stdout,
            'error': result.stderr,
            'success': result.returncode == 0
        })
    except Exception as e:
        return jsonify({'error': str(e)}), 500

if __name__ == '__main__':
    app.run(debug=True, port=5000)

    warning: /WARNING: (.+)/g

function extractErrors(terminalOutput) {
  const results = {
    errors: [],
    warnings: [],
    file_references: [],
    stack_trace: null
  };
  
  // Auto-detect error type and extract relevant info
  Object.entries(errorPatterns).forEach(([lang, patterns]) => {
    Object.entries(patterns).forEach(([type, regex]) => {
      const matches = [...terminalOutput.matchAll(regex)];
      if (matches.length > 0) {
        results.errors.push({
          language: lang,
          type: type,
          matches: matches.map(m => m[1])
        });
      }
    });
  });
  
  return results;
}

        function parseResumeData() {
            const rawData = document.getElementById('pasteArea').value.trim();
            if (!rawData) {
                alert('Please paste some resume data first!');
                return;
            }

            const batchMarker = '=== RESUME ';
            const isBatchFormatDetected = rawData.includes(batchMarker);
            let resumeBlocks = [];

            if (isBatchFormatDetected) {
                resumeBlocks = rawData.split(/=== RESUME \d+ ===/);
                resumeBlocks = resumeBlocks.filter(block => block.trim()); // Remove empty strings
            }

            if (isBatchFormatDetected && resumeBlocks.length > 1) {
                // True batch processing scenario (multiple === RESUME X === blocks)
                processBatchData(resumeBlocks); // Pass the already split blocks
            } else if (isBatchFormatDetected && resumeBlocks.length === 1) {
                // Single resume wrapped in === RESUME 1 === format, but not a true batch
                resumeData = parseSingleResume(resumeBlocks[0]);
                validateData();
                renderValidation();
            } else {
                // No batch markers found, treat as a single resume
                resumeData = parseSingleResume(rawData);
                validateData();
                renderValidation();
            }
        }

        function parseSingleResume(data) {
            const parsed = {};
            const lines = data.split('\n');
            let currentSection = '';
            let jobs = [];
            let currentJob = null;
            let metrics = [];
            let skills = [];

            for (let line of lines) {
                line = line.trim();
                if (!line) continue;

                // Main fields at the top, until a section header is hit
                if (line.includes(':') && !line.startsWith('-') && !currentSection) {
                    const [key, ...valueParts] = line.split(':');
                    parsed[key.trim()] = valueParts.join(':').trim();
                }

                // Section headers
                if (line === 'EXPERIENCE:') {
                    currentSection = 'experience';
                    // Reset currentJob for the new section
                    currentJob = null;
                }
                else if (line === 'METRICS:') currentSection = 'metrics';
                else if (line === 'SKILLS:') currentSection = 'skills';
                
                // Parsing content within sections
                else if (currentSection === 'experience') {
                    if (line.startsWith('Job Title:')) {
                        if (currentJob) jobs.push(currentJob); // Push previous job if exists
                        currentJob = { title: line.replace('Job Title:', '').trim(), company: '', achievements: [] };
                    } else if (line.startsWith('Company:')) {
                        if (currentJob) currentJob.company = line.replace('Company:', '').trim();
                    } else if (line.startsWith('-')) {
                        if (currentJob) currentJob.achievements.push(line.substring(1).trim());
                    }
                } else if (currentSection === 'metrics' && line.includes('|')) {
                    metrics.push(line);
                } else if (currentSection === 'skills' && line.includes('|')) {
                    skills.push(line);
                }
            }

            if (currentJob) jobs.push(currentJob); // Push the last job if it exists
            
            parsed['Experience Jobs'] = jobs;
            parsed['Metrics'] = metrics;
            parsed['Skills'] = skills;

            return parsed;
        }

        function processBatchData(parsedResumeBlocks) { // Accepts array of resume content blocks

                alert('Error: Batch processing requires more than one resume block.');
                return;
            }

            const prefix = prompt(`Enter subdomain prefix for ${parsedResumeBlocks.length} resumes:`);
            if (!prefix) return;

            let generatedHTML = '';
            let batchResumeCount = 0;
            parsedResumeBlocks.forEach((block, idx) => {
                const data = parseSingleResume(block.trim());
                const subdomain = `${prefix}${idx + 1}`;
                const html = generateResumeHTML(data, subdomain);
                
                generatedHTML += `<!-- ===== ${subdomain}.trevino.today ===== -->\n`;

            profiles[currentProfile].resumeCount += batchResumeCount; // Update for current profile
            updateResumeCount();
            showCopySuccess(`Generated ${batchResumeCount} resumes!`);
        }

        function validateData() {
            validationState = {};

            for (let field of requiredFields) {
                const value = resumeData[field];
                validationState[field] = {
                    status: value && value.trim() ? 'valid' : 'missing',
                    value: value || '',
                    approved: false
                };
            }

            const jobs = resumeData['Experience Jobs'] || [];
            validationState['Experience Jobs'] = {
                status: jobs.length > 0 ? 'valid' : 'missing',
                value: jobs,
                approved: false
            };

            const metrics = resumeData['Metrics'] || [];
            validationState['Metrics'] = {
                status: metrics.length >= 3 ? 'valid' : 'missing', // Adjusted to 3 for provided example
                value: metrics,
                approved: false
            };

            const skills = resumeData['Skills'] || [];
            validationState['Skills'] = {
                status: skills.length >= 1 ? 'valid' : 'missing', // Adjusted to 1 for provided example
                value: skills,
                approved: false
            };
        }

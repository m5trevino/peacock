class LegalResearchMCP:
    def __init__(self):
        self.client = chromadb.PersistentClient(
            path="./legal_chroma_db",
            settings=Settings(allow_reset=True)
        )
        self.eviction_collection = None
        self._ensure_collections()
    
    def _ensure_collections(self):
        """Create eviction research collection if it doesn't exist"""
        try:
            self.eviction_collection = self.client.get_collection("eviction_research")
        except:
            self.eviction_collection = self.client.create_collection(
                name="eviction_research",
                metadata={"description": "Legal documents and research for eviction case"}
            )
    
    async def add_document(self, content: str, doc_type: str, metadata: Dict[str, Any] = None):
        """Add legal document to research collection"""
        if metadata is None:
            metadata = {}
        
        metadata.update({
            "doc_type": doc_type,
            "content_length": len(content)
        })
        
        # Generate unique ID
        doc_id = f"{doc_type}_{len(content)}_{hash(content) % 10000}"
        
        self.eviction_collection.add(
            documents=[content],
            metadatas=[metadata],
            ids=[doc_id]
        )
        
        return {"status": "added", "id": doc_id, "type": doc_type}
    
    async def search_documents(self, query: str, n_results: int = 5):
        """Search through legal research documents"""
        results = self.eviction_collection.query(
            query_texts=[query],
            n_results=n_results,
            include=["documents", "metadatas", "distances"]
        )
        
        return {
            "query": query,
            "results": [
                {
                    "document": doc,
                    "metadata": meta,
                    "relevance_score": 1 - distance
                }
                for doc, meta, distance in zip(
                    results["documents"][0],
                    results["metadatas"][0], 
                    results["distances"][0]
                )
            ]
        }
    
    async def get_collection_stats(self):
        """Get stats about the research collection"""
        count = self.eviction_collection.count()
        return {
            "total_documents": count,
            "collection_name": "eviction_research"
        }

# MCP Protocol Implementation
async def handle_mcp_request(request_data):
    mcp = LegalResearchMCP()
    
    method = request_data.get("method")
    params = request_data.get("params", {})
    
    if method == "add_document":
        return await mcp.add_document(
            content=params.get("content"),
            doc_type=params.get("doc_type", "general"),
            metadata=params.get("metadata", {})
        )
    
    elif method == "search_documents":
        return await mcp.search_documents(
            query=params.get("query"),
            n_results=params.get("n_results", 5)
        )
    
    elif method == "get_stats":
        return await mcp.get_collection_stats()
    
    else:
        return {"error": f"Unknown method: {method}"}

async def main():
    while True:
        try:
            line = input()
            request = json.loads(line)
            result = await handle_mcp_request(request)

            sys.stdout.flush()
        except EOFError:
            break
        except Exception as e:
            print(json.dumps({"error": str(e)}))
            sys.stdout.flush()

import os
import chromadb
from chromadb.config import Settings
import PyPDF2

class LegalContentLoader:
    def __init__(self):
        self.client = chromadb.PersistentClient(
            path="./legal_chroma_db",
            settings=Settings(allow_reset=True)
        )
        self.collection = self._get_or_create_collection()
    
    def _get_or_create_collection(self):
        try:
            return self.client.get_collection("eviction_research")
        except:
            return self.client.create_collection(
                name="eviction_research",
                metadata={"description": "Eviction case legal research documents"}
            )
    
    def load_pdf(self, filepath):
        """Extract text from PDF"""
        try:
            with open(filepath, 'rb') as file:
                reader = PyPDF2.PdfReader(file)
                text = ""
                for page in reader.pages:
                    text += page.extract_text() + "\n"
                return text
        except Exception as e:
            print(f"Error reading PDF {filepath}: {e}")
            return None
    
    def load_markdown(self, filepath):
        """Load markdown file"""
        try:
            with open(filepath, 'r', encoding='utf-8') as file:
                return file.read()
        except Exception as e:
            print(f"Error reading MD {filepath}: {e}")
            return None
    
    def chunk_content(self, content, chunk_size=1000, overlap=200):
        """Split content into overlapping chunks"""
        if not content or len(content) < chunk_size:
            return [content] if content else []
        
        chunks = []
        start = 0
        while start < len(content):
            end = start + chunk_size
            chunk = content[start:end]
            chunks.append(chunk)
            start = end - overlap
        return chunks
    
    def load_content_directory(self, content_dir):
        """Load all content from directory"""
        content_path = Path(content_dir)
        loaded_count = 0
        
        for file_path in content_path.iterdir():
            if file_path.is_file():
                print(f"Processing: {file_path.name}")
                
                # Determine file type and extract content
                content = None
                if file_path.suffix.lower() == '.pdf':
                    content = self.load_pdf(file_path)
                    doc_type = "pdf"
                elif file_path.suffix.lower() == '.md':
                    content = self.load_markdown(file_path)
                    doc_type = "markdown"
                else:
                    continue
                
                if content:
                    # Chunk the content
                    chunks = self.chunk_content(content)
                    
                    # Add each chunk to collection
                    for i, chunk in enumerate(chunks):
                        doc_id = f"{file_path.stem}_chunk_{i}"
                        metadata = {
                            "filename": file_path.name,
                            "doc_type": doc_type,
                            "chunk_index": i,
                            "total_chunks": len(chunks),
                            "file_size": file_path.stat().st_size
                        }
                        
                        self.collection.add(
                            documents=[chunk],
                            metadatas=[metadata],
                            ids=[doc_id]
                        )
                    
                    loaded_count += 1
                    print(f"âœ“ Loaded {len(chunks)} chunks from {file_path.name}")
        
        print(f"\nğŸ”¥ LOADED {loaded_count} files into ChromaDB")
        return loaded_count

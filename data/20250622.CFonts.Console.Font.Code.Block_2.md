#!/usr/bin/env python3

import subprocess
import json

def run_cfonts(text, font='block', colors=None, gradient=None, background='transparent', align='center'):
    """Bridge to call cfonts from Python"""
    
    cmd = ['npx', 'cfonts', text, '-f', font, '--align', align]
    
    if gradient:
        cmd.extend(['-g', ','.join(gradient)])
    elif colors:
        cmd.extend(['-c', colors[0] if isinstance(colors, list) else colors])
    
    if background != 'transparent':
        cmd.extend(['-b', background])
    
    try:
        result = subprocess.run(cmd, capture_output=True, text=True)
        print(result.stdout)
    except FileNotFoundError:
        print("cfonts not available - install with: npm install -g cfonts")

# Test it
if __name__ == "__main__":
    print("=== PYTHON CALLING CFONTS ===")
    
    # Some examples
    run_cfonts("PEACOCK", font='block', gradient=['red', 'yellow'])
    run_cfonts("MEMORY", font='slick', gradient=['green', 'cyan'])
    run_cfonts("TEST", font='chrome', colors=['magenta'])

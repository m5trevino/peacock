            searchDocuments(query, filters) {
                let results = [...this.documents];

                // Apply text search
                if (query) {
                    const queryLower = query.toLowerCase();
                    results = results.filter(doc => {
                        return (
                            doc.name.toLowerCase().includes(queryLower) ||
                            (doc.ai_summary && doc.ai_summary.toLowerCase().includes(queryLower)) ||
                            (doc.key_terms && doc.key_terms.some(term => term.toLowerCase().includes(queryLower))) ||

#!/usr/bin/env python3
"""
Ollama MCP Client - connects your Peacock Memory to Ollama
"""
import requests
import json
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client

# Your Peacock MCP server config
PEACOCK_MCP_CONFIG = {
    "peacock-memory": {
        "command": "/home/flintx/.pyenv/versions/peacock/bin/python",
        "args": ["/home/flintx/peacock-mem/mcp_server_proper.py"]
    }
}

OLLAMA_URL = "http://127.0.0.1:11434"

async def chat_with_ollama_and_mcp(prompt: str, model: str = "qwen2.5:7b"):
    # Connect to your Peacock Memory MCP server
    server_params = StdioServerParameters(
        command=PEACOCK_MCP_CONFIG["peacock-memory"]["command"],
        args=PEACOCK_MCP_CONFIG["peacock-memory"]["args"]
    )
    
    async with stdio_client(server_params) as (read, write):
        async with ClientSession(read, write) as session:
            # Initialize MCP session
            await session.initialize()
            
            # Search your memory for relevant context
            search_result = await session.call_tool("search_memory", {"query": prompt})
            
            # Build enhanced prompt with your memory context
            context = search_result.content[0].text if search_result.content else ""
            enhanced_prompt = f"Context from memory:\n{context}\n\nUser query: {prompt}"
            
            # Send to Ollama
            response = requests.post(f"{OLLAMA_URL}/api/generate", json={
                "model": model,
                "prompt": enhanced_prompt,
                "stream": False
            })
            
            return response.json()

if __name__ == "__main__":
    import asyncio
    
    prompt = input("Enter your prompt: ")
    result = asyncio.run(chat_with_ollama_and_mcp(prompt))
    print(result.get("response", "No response"))
EOF
